# 2 移除元素

## [27 移除元素](https://leetcode.cn/problems/remove-element/description/)

```
class Solution {

public:

    int removeElement(vector<int>& nums, int val) {

        int fastptr = 0;

        int lowptr = 0;

        int len = nums.size();

        for(;fastptr <nums.size();){

            if(nums[fastptr]!=val){

                nums[lowptr]=nums[fastptr];

                fastptr++;

                lowptr++;

            }

            else{

                fastptr++;

                len--;

            }

        }

        return len;

    }

};
```
