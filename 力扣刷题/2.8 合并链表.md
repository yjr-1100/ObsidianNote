# 2.8 合并链表

[21. 合并两个有序链表](https://leetcode.cn/problems/merge-two-sorted-lists/)

```cpp
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        ListNode* head=new ListNode(0);
        ListNode* curr = head;
        while(list1!=nullptr&&list2!=nullptr){
            if(list1->val<list2->val){
                curr->next = list1;
                list1 = list1->next;
            }else{
                curr->next = list2;
                list2=list2->next;
            }
            curr = curr->next;
        }
        if(list1!=nullptr) curr->next = list1;
        if(list2!=nullptr) curr->next = list2;
        return head->next;
    }
};
```

## [23. 合并 K 个升序链表](https://leetcode.cn/problems/merge-k-sorted-lists/)

直接每次比较所有链表的头，哪个小就插入最后的链表中，时间复杂度比较大 假设一每个链表都有n个节点，k个链表，则复杂度是O(n k^2)

```cpp
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        ListNode* head = new ListNode(0);
        ListNode* ans = head;
        while(lists.size()>1){
            int min = INT_MAX;
            auto curr = lists.begin();
            for(auto it = lists.begin(); it != lists.end();){
                if((*it)==nullptr){
                    it = lists.erase(it);
                }
                else{
                    if(min>(*it)->val){
                        min = (*it)->val;
                        curr = it;
                    }
                    it++;
                }
            }
            if(!lists.empty()){
                head->next = (*curr);
                head = head->next;
                (*curr) = (*curr)->next;
            }
        }
        if(!lists.empty()&&lists[0]!=nullptr)
            head->next = lists[0];
        return ans->next;
    }
};
```

可以对链表两两合并，这样就不需要每次遍历k去找哪个最小了。时间复杂度是O(kn * logk); 

```cpp
class Solution {
private:
    ListNode* mergeTwoLists(ListNode* a, ListNode* b){
        ListNode dummy(-1);
        ListNode *head = &dummy, *p = head;
        while(a&&b){
            if(a->val<b->val){
                p->next = a;
                a = a->next;
            } else {
                p->next = b;
                b = b->next;
            }
            p = p->next;
        }
        if(a) p->next = a;
        else if(b) p->next = b;
        return head->next;
    }
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        if(lists.empty()) return NULL;
        
        int lsize = lists.size();
        while(lsize>1){
            for(int i=0;i<lsize/2;++i){
                lists[i] = mergeTwoLists(lists[i], lists[i+(lsize+1)/2]);
            }
            lsize = (lsize+1)/2;
        }
        return lists[0];
    }
};
```


## [86. 分隔链表](https://leetcode.cn/problems/partition-list/)

```cpp
class Solution {
public:
    ListNode* partition(ListNode* head, int x) {

        head = new ListNode(0,head);

        ListNode* fast = head;

        ListNode* slow = head;

        while(fast!=nullptr&&fast->next!=nullptr){

            if(fast->next->val<x&&fast!=slow){

                ListNode* t = fast->next;

                fast->next = t->next;

                t->next = slow->next;

                slow->next = t;

                slow=slow->next;

            }

            else if(fast->next->val<x&&fast==slow){

                fast = fast->next;

                slow = slow->next;

            }

            else{

                fast = fast->next;

            }

        }

        return head->next ;

    }

};
```
