# 2.2 设计链表

  [707. 设计链表](https://leetcode.cn/problems/design-linked-list/)

```cpp
class MyLinkedList {
private:
    struct LinkNode{
        int val;
        LinkNode* next;
        LinkNode(int v): val(v),next(nullptr){}
    };
    LinkNode* head;
public:
    MyLinkedList() {
        head = new LinkNode(0);
    }
    int get(int index) {
        int ans = -1;
        int i = 0;
        LinkNode* curr = head->next;
        while(curr!=nullptr){
            if(i==index){
                ans = curr->val;
                break;
            }
            curr = curr->next;
            i++;
        }
        return ans;
    }
    void addAtHead(int val) {
        LinkNode* a = new LinkNode(val);
        a->next = head->next;
        head->next = a;
    }
    void addAtTail(int val) {
        LinkNode* a = new LinkNode(val);
        LinkNode* curr = head;
        while(curr->next!=nullptr) curr = curr->next;
        curr->next = a;
    }
    void addAtIndex(int index, int val) {
        LinkNode* a = new LinkNode(val);
        LinkNode* curr = head;
        int i = 0;
        while(curr->next!=nullptr){
            if(i==index){
                a->next = curr->next;
                curr->next = a;
                i++;
                break;
            }
            curr = curr->next;
            i++;
        }
        if(i==index) curr->next = a;
    }
    void deleteAtIndex(int index) {
        LinkNode* curr = head;
        int i = 0;
        while(curr!=nullptr&&curr->next!=nullptr){
            if(i==index){
                LinkNode* a = curr->next;
                curr->next = a->next;
                delete a;
            }
            curr = curr->next;
            i++;
        }
    }
};

 * int param_1 = obj->get(index);
 * obj->addAtHead(val);
 * obj->addAtTail(val);
 * obj->addAtIndex(index,val);
 * obj->deleteAtIndex(index);
```
